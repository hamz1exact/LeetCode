```python
class Solution:
    def threeSum(self, nums: List[int]) -> List[List[int]]:
        arr = []
        nums.sort()
        for x in range(len(nums)):
            if nums[x] > 2:
                break
            if x > 0 and nums[x] == nums[x-1]:
                continue
            r = len(nums) - 1
            l = x + 1
            while l < r:
                ThreeSum = nums[x] + nums[l] + nums[r]
                if ThreeSum > 0:
                    r -=1
                elif ThreeSum < 0:
                    l +=1
                elif ThreeSum == 0:
                    arr.append([nums[x],  nums[l] , nums[r]])
                    l += 1
                    r -= 1
                    while nums[l] == nums[l-1] and l < r:
                        l += 1                        
        return arr
```
